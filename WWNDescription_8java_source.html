7 /**
8  * @file
9  * In situations where nicknames simply are not present, but we need descriptors in short-order,
10  * the "--wwn=" or "-w" ooption can be used to get a description of what the most likely Nickname
11  * would be. For example, "java -jar wwndesc.jar -w 5006048ACCC86A32" results in "Symm-182500953-05bA"
12  * matching http://www.emcstorageinfo.com/2007/08/how-to-decode-symmetrix-world-wide.html
13  * Note that --briefestimate and --nobriefestimate configure for more brief estimate/suggested nicknames
14  *
15  * @section Examples
16  *
17  * java -jar vict.jar --wwn 500610601234567
18  *
19  * java -jar vict.jar --briefestimate --wwn 500610601234567
20  *
21  * java -jar vict.jar --nobriefestimate --wwn 5000097301234564
22  *
23  * @section Known Issues
24  */

116  /**
117  * Main function, as you can tell.
118  *
119  * this function merely parses the command-line to dispatch actions to the meat of the meal above.
120  * I'm using an actual GetOpt because, yes, I'm a UNIX/C hack, re-using 3-decades-old knowledge,
121  * but this also preserves both extensibility and the ability to use longopts in scripts as a
122  * way to self-document what the tool's doing.
123  *
124  * No real intelligence herein except the parse-and-redirect action.
125  *
126  * @param args commandline arguments passed in by the JRE
127  */
128  public static void main(String args[])
129  {
130  WWNDescription m = new WWNDescription();
131 
132  java.util.Vector<LongOpt> options = new java.util.Vector();
133 
134  /* Always always ALWAYS provide a quick reference and a version output */
135  options.add(new LongOpt("help", LongOpt.NO_ARGUMENT, null, 'H'));
136  options.add(new LongOpt("version", LongOpt.NO_ARGUMENT, null, 'V'));
137 
138  options.add(new LongOpt("wwn", LongOpt.REQUIRED_ARGUMENT, null, 'w'));
139  options.add(new LongOpt("briefestimate", LongOpt.NO_ARGUMENT, null, 0x1000));
140  options.add(new LongOpt("nobriefestimate", LongOpt.NO_ARGUMENT, null, 0x1001));
141 
142  org.smallfoot.getopt.GetOpt g = new org.smallfoot.getopt.GetOpt("wwndesc", args, options);
143 
144  int c;
145  while ((c = g.getopt()) != -1)
146  {
147  switch(c)
148  {
149  case 0x1000: /* switch to brief WWNDescs */
150  m.briefWWNEstimate = true;
151  break;
152  case 0x1001: /* switch to verbose WWNDescs */
153  m.briefWWNEstimate = false;
154  break;
155  case 'w': /* inspect a WWN */
156  {
157  WWNDesc wwn;
158  if ((null == g.getOptarg()) || (16 != g.getOptarg().replaceAll(":","").length()))
159  System.out.println("no result: parameter "+g.getOptarg()+" needs to be a 16-digit hexadecimal");
160  else if (null == (wwn = m.getWWNDescriptor(g.getOptarg())))
161  System.out.println("no result: WWN "+g.getOptarg()+" has no descriptor");
162  else System.out.println(wwn.toString());
163  }
164  break;
165 
166  /*
167  * Follows is the "house-keeping": versions, usage, and the catch-all for bad options.
168  */
169  case 'V': // print the version and quit
170  {
171  System.out.println(g.consistentVersion("1.0-Revision 631 $")+"\n");
172  return;
173  }
174 
175  default:
176  case '?':
177  // during build, this is just a dump of options; in shipping, this falls-thru to usage.
178  //System.out.println("option \""+c+"\" selected");
179  //System.out.println("long index = "+g.getLongind());
180 
181  case 'H':
182  m.usage(g.progname());
183  break;
184  }
185  }
186 
187  /*
188  * I had to add this since there seems to be a lack of dropping out when the options are done,
189  * or a delay huge enough that my impatience is triggered (ie 2 seconds or more!). Maybe
190  * invesitgate this further when there's more time, as this is redundant.
191  */
192  java.lang.System.exit(0);
193  }
194 }
